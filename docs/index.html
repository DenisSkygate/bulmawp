function bulmawp_get_pagination_config() {
  $paginationAlignment = get_option( 'pagination_alignment' );
  if( $paginationAlignment == 'is-centered' ) {
    echo 'is-centered';
  } else if( $paginationAlignment == 'is-right' ) {
    echo 'is-right';
  }
}
